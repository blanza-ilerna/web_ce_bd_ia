Arquitectura de un Sistema Basado en Reglas

üìö Base de Conocimientos

                continue
        
        return self.conclusiones
    
    def explicar_razonamiento(self):
        """Proporciona trazabilidad del proceso de inferencia"""
        print("\nüìã Hechos conocidos:")
        for clave, valor in self.hechos.items():
            print(f"   ‚Ä¢ {clave}: {valor}")
        
        print(f"\nüéØ Conclusiones alcanzadas ({len(self.conclusiones)}):")
