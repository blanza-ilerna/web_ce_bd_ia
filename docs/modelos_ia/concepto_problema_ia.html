Definici√≥n del Problema

Formalizaci√≥n del Problema:

def es_seguro(tablero, fila, col):
    """
    Verifica si es seguro colocar una reina en tablero[fila][col]
    Comprueba restricciones: misma columna, diagonal izq, diagonal der
    """
    # Verificar columna hacia arriba
    for i in range(fila):
        if tablero[i][col] == 1:
            return False

    # Verificar diagonal superior izquierda
    i, j = fila - 1, col - 1
    while i >= 0 and j >= 0:
        if tablero[i][j] == 1:
            return False
        i -= 1
        j -= 1

    # Verificar diagonal superior derecha
    i, j = fila - 1, col + 1
    while i >= 0 and j < len(tablero):
        if tablero[i][j] == 1:
            return False
        i -= 1
        j += 1

    return True

def resolver_n_reinas_util(tablero, fila):
    """
    Utiliza BACKTRACKING para resolver el problema
    - FASE 1 (Formulaci√≥n): Ya definida arriba
    - FASE 2 (Representaci√≥n): Tablero = matriz 8x8
    - FASE 3 (B√∫squeda): Backtracking DFS
    - FASE 4 (Ejecuci√≥n): Devuelve soluci√≥n encontrada
    """
    n = len(tablero)

    # Caso base: todas las reinas colocadas (E_f alcanzado)
    if fila >= n:
        return True

    # Probar colocar reina en cada columna de esta fila
    for col in range(n):
        # Verificar RESTRICCIONES (R)
        if es_seguro(tablero, fila, col):
            # Aplicar ACCI√ìN: colocar reina
            tablero[fila][col] = 1

            # Recursi√≥n: intentar colocar resto de reinas
            if resolver_n_reinas_util(tablero, fila + 1):
                return True

            # BACKTRACKING: si no funcion√≥, quitar reina
            tablero[fila][col] = 0

    # Si no hay soluci√≥n en esta rama
    return False

def resolver_n_reinas(n=8):
    """
    Funci√≥n principal: inicializa y resuelve el problema
    """
    # FASE 2: Representaci√≥n (matriz nxn)
    tablero = [[0 for _ in range(n)] for _ in range(n)]

    # FASE 3: B√∫squeda
    if resolver_n_reinas_util(tablero, 0):

        imprimir_tablero(tablero)
        return tablero
    else:
        print(f"No existe soluci√≥n para {n} reinas")
        return None

    for fila in tablero:
        print(" ".join("Q" if celda == 1 else "." for celda in fila))
