1. B√∫squeda en Anchura (BFS - Breadth-First Search)

from collections import deque

def busqueda_anchura(grafo, inicio, meta):
    """
    B√∫squeda en Anchura (BFS)
    Explora nivel por nivel usando una cola FIFO
    """
    cola = deque([(inicio, [inicio])])  # (nodo_actual, camino)
    visitados = set([inicio])
    
    while cola:
        nodo_actual, camino = cola.popleft()
        
        # ¬øHemos llegado a la meta?
        if nodo_actual == meta:
            return camino
        
        # Explorar vecinos
        for vecino in grafo[nodo_actual]:
            if vecino not in visitados:
                visitados.add(vecino)
                cola.append((vecino, camino + [vecino]))
    
    return None  # No hay soluci√≥n

def busqueda_profundidad(grafo, inicio, meta, visitados=None, camino=None):

    B√∫squeda en Profundidad (DFS) - Implementaci√≥n recursiva

El algoritmo A* es el est√°ndar de oro en b√∫squeda informada. Combina el coste recorrido con una estimaci√≥n heur√≠stica del coste restante mediante la funci√≥n de evaluaci√≥n:

‚úÖ Propiedades de A*:

import heapq

def a_estrella(grafo, inicio, meta, heuristica):
    """
    Algoritmo A*: B√∫squeda informada √≥ptima
    f(n) = g(n) + h(n)
    """
    # Cola de prioridad: (f_score, g_score, nodo, camino)
    frontera = [(heuristica(inicio, meta), 0, inicio, [inicio])]
    explorados = set()
    
    while frontera:
        f_score, g_score, nodo_actual, camino = heapq.heappop(frontera)
        
        # ¬øHemos llegado a la meta?
        if nodo_actual == meta:
            return camino, g_score
        
        if nodo_actual in explorados:
            continue
        
        explorados.add(nodo_actual)
        
        # Explorar vecinos
        for vecino, coste in grafo[nodo_actual].items():
            if vecino not in explorados:
                nuevo_g = g_score + coste
                nuevo_h = heuristica(vecino, meta)
                nuevo_f = nuevo_g + nuevo_h
                nuevo_camino = camino + [vecino]
                heapq.heappush(frontera, (nuevo_f, nuevo_g, vecino, nuevo_camino))
    
    return None, float('inf')  # No hay soluci√≥n
